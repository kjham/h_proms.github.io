function fetchData() {
	// 1. 변수 생성
	var result = [];
	// 2. 데이터 요청
	$.ajax({
		url: 'https://api.hnpwa.com/v0/news/1.json',
		success: function(data) {
			console.log('데이터 호출 결과', data);
			result = data;
		}	
	});
	// 3. 결과 출력
	console.log('함수 결과', result);
}

출력 결과는 다음과 같이 나옵니다..

1
2

함수 결과: []
데이터 호출 결과 : [{ ... }]

이것은 $.ajax가 비동기로 처리되기 때문에 발생하는 현상입니다.
따라서, CallBack 함수 내 데이터 처리 로직이 작성되면 코드가 복잡하게 됩니다.
이에 대응하여 나온 것은 Promise 입니다.

function callAjax() {
	return new Promise(function(resolve, reject) {
		$.ajax({
			url: 'https://api.hnpwa.com/v0/news/1.json',
			success: function(data) {
				resolve(data);
			}
		});
	})
}
function fetchData() {
	// 1. 변수 생성
	var result = [];
	// 2. 데이터 요청
	callAjax()
		.then(function(data) {
			// 3. 결과 출력
			console.log('데이터 호출 결과', data);
			result = data;
			console.log('함수 결과', result);
		});
}
